### 3) Señal y[n] resultante de la convolución usando Python
´´´ python 

x = np.array([1,1,2,3,4,3,3,4,0,2])
h = np.array([5,6,0,0,8,6,9])

y = np.convolve(x, h)  # convolución discreta
print("y[n] =", y)

n = np.arange(len(y))

plt.stem(n, y)
plt.xlabel("n")
plt.ylabel("y[n]")
plt.title("Convolución discreta")
plt.grid(True)
plt.show()
